CREATE TABLE IF NOT EXISTS employees (
    employee_id TEXT PRIMARY KEY,
    first_name TEXT NOT NULL,
    last_name TEXT NOT NULL,
    email TEXT UNIQUE NOT NULL,
    department_id INTEGER NOT NULL,
    
    -- Leave Balance Tracking
    vacation_allocated INTEGER NOT NULL DEFAULT 0,
    vacation_used INTEGER NOT NULL DEFAULT 0,
    sick_allocated INTEGER NOT NULL DEFAULT 0,
    sick_used INTEGER NOT NULL DEFAULT 0,
    
    -- Relationship linking back to the Department table
    FOREIGN KEY (department_id) REFERENCES departments(department_id)
);

CREATE TABLE IF NOT EXISTS departments (
    department_id INTEGER PRIMARY KEY AUTOINCREMENT,
    department_name TEXT NOT NULL UNIQUE
);

CREATE TABLE IF NOT EXISTS schedules (
    shift_id INTEGER PRIMARY KEY AUTOINCREMENT,
    employee_id TEXT NOT NULL,
    shift_date TEXT NOT NULL,
    start_time TEXT NOT NULL,
    end_time TEXT NOT NULL,
    
    -- Relationship linking back to the Employee table
    FOREIGN KEY (employee_id) REFERENCES employees(employee_id)
);

CREATE TABLE IF NOT EXISTS audit_logs (
    log_id INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp TEXT NOT NULL,
    employee_id TEXT,
    user_message TEXT NOT NULL,
    classified_intent TEXT NOT NULL,
    confidence_score REAL NOT NULL,
    final_response TEXT NOT NULL,
    
    -- Optional relationship linking back to the Employee table
    FOREIGN KEY (employee_id) REFERENCES employees(employee_id)
);
